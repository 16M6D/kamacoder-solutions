
# 14. 句子缩写 

[题目链接](https://kamacoder.com/problem.php?id=1013) 


## C++ 

```CPP 
#include<iostream>
#include<string>
using namespace std;
char changeChar(char & a) { // 小写变大写
    if (a >= 'a' && a <= 'z') a -= 32;
    return a;
}
int main() {
    int n;
    char tmp;
    string result, s;
    cin >> n;
    tmp = getchar(); // 吸收一个回车，因为输入n之后，要输入一个回车
    while (n--) {
        result = "";
        getline(cin, s);
        result += changeChar(s[0]);
        for (int i = 1; i < s.size() - 1; i++) { // 单词之间可能有多个空格
            if (s[i] == ' ' && s[i + 1] != ' ') result += changeChar(s[i + 1]);
        }
        cout << result << endl;
    }
}
```
