## C++ 

```CPP 
#include<iostream>
using namespace std;
int main() {
    int n, a, b;
    while (cin >> n) {
        while (n--) {
            cin >> a;
            int sum = 0;
            while (a--) {
                cin >> b;
                sum += b;
            }
            cout << sum << endl;
            if (n != 0) cout << endl;
        }
    }
}
```
## Java 

```Java
import java.util.Scanner;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        while(sc.hasNextLine()){
            int N = sc.nextInt();
            while(N-- > 0){
                int M = sc.nextInt();
                int sum = 0;
                while(M-- > 0){
                    sum += sc.nextInt();
                }
                System.out.println(sum);
                if(N > 0) System.out.println();
            }
        }
 
    }
}
```
