
# 8. 摆平积木 

[题目链接](https://kamacoder.com/problem.php?id=1007) 

## C++ 

```CPP 
#include<iostream>
#include<vector>
using namespace std;

int main() {
    int n;
    while (cin >> n) {
        if (n == 0) break;
        vector<int> nums = vector<int>(n, 0);
        int sum = 0;
        for (int i = 0; i < n; i++) {
            cin >> nums[i];
            sum += nums[i];
        }
        int average = sum / n;
        int result = 0;
        for (int i = 0; i < n; i++) {
            if ((nums[i] - average) > 0) result += (nums[i] - average);
        }
        cout << result << endl;
        cout<< endl;
    }
}
```

## python 

```python
while 1:
    try:
        n = int(input())
        if n == 0:
            break
        ls = list(map(int, input().split()))
        ls.sort()
        m = sum(ls) // n
        moves = 0
        for i in ls:
            moves += abs(i - m)
        print(moves//2)
        print()
    except:
        break
```
