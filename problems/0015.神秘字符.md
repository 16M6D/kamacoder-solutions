
# 15. 神秘的字符 

[题目链接](https://kamacoder.com/problem.php?id=1014) 

## C++

```CPP  
#include<iostream>
#include<string>
using namespace std;
int main() {
    int n;
    char tmp;
    cin >> n;
    string s, t;
    while (n--) {
        cin >> s;
        tmp = getchar(); // 吸收输入中，一个回车
        cin >> t;
        string result = "";
        for (int i = 0; i < s.size() / 2; i++) { // 处理s的前半段
            result += s[i];
        }
        result += t; // 添加 字符串t
        for (int i = s.size() / 2; i < s.size(); i++) { // 处理s的后半段
            result += s[i];
        }
        cout << result << endl;
    }
}

```
