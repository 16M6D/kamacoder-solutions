
# 13. 镂空三角形 

[题目链接](https://kamacoder.com/problem.php?id=1012) 

## C++ 

```CPP 
#include<iostream>
using namespace std;
void printSpaces(int numSpaces) {
    for (int i = 0; i < numSpaces; ++i) {
        cout << " ";
    }
}
void printAChars(int numChars, char a) {
    for (int i = 0; i < numChars; ++i) {
        cout << a;
    }
}
int main() {
    int n;
    char a;
    while (cin >> a) {
        if (a == '@') break;
        cin >> n;
        if (n == 1) cout << a << endl;
        for (int i = 1; i <= n - 1; ++i) {
            // 打印空格
            printSpaces(n - i);
             // 打印字符
            if (i == 1) {
                cout << a;
            } else {
                cout << a;
                printSpaces(2 * i - 3);
                cout << a;
            }
            cout << endl;
        }
         // 打印底边
        if (n > 1) {
            printAChars(2 * n - 1, a);
            cout << endl;
        }
        cout << endl; // 每条结果下一个空行
    }
}

```
